CREATE TABLE members(

	m_no VARCHAR2(100) PRIMARY KEY,
	m_email VARCHAR2(100),
	m_pwd VARCHAR2(100),
	m_name VARCHAR2(100),
	m_birthday VARCHAR2(100)
	
);

CREATE SEQUENCE members_mno_seq;
-----------------------------------
CREATE TABLE boards(

	b_id VARCHAR2(100) PRIMARY KEY,
	b_writer VARCHAR2(100),		
	b_title VARCHAR2(100),			
	b_content VARCHAR2(100),			
	b_pwd VARCHAR2(100)
	
);

CREATE SEQUENCE boards_bid_seq;
